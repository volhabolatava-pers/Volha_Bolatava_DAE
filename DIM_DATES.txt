Create Table BL_DM.DIM_DATES_DAY (
	DATE_ID    BIGINT PRIMARY KEY, 
    	DATE_DT    DATE NOT NULL,      
	"DAY"     INTEGER NOT NULL,
  	DAY_OF_WEEK  VARCHAR(10) NOT NULL,
  	IS_WEEKEND   Boolean NOT NULL,  
    	"MONTH"   INTEGER NOT NULL,   
    	"QUARTER" INTEGER NOT NULL,   
    	"YEAR"    INTEGER NOT NULL
);

INSERT INTO BL_DM.DIM_DATES_DAY (
    	DATE_ID, 
    	DATE_DT, 
	"DAY",
	DAY_OF_WEEK,
	IS_WEEKEND,
    	"MONTH", 
    	"QUARTER", 
    	"YEAR"
)
SELECT
	CAST(TO_CHAR(datum, 'YYYYMMDD') AS BIGINT) AS DATE_ID,
    	datum DATE_DT,
	EXTRACT(DAY FROM datum) "DAY",
	TRIM(TO_CHAR(datum, 'Day')) DAY_OF_WEEK,
	(EXTRACT(ISODOW FROM datum) > 5) AS IS_WEEKEND,
	EXTRACT(MONTH FROM datum) "MONTH",
   	EXTRACT(QUARTER FROM datum) "QUARTER",
    	EXTRACT(YEAR FROM datum) "YEAR"
FROM GENERATE_SERIES('2010-01-01'::DATE,'2030-12-31'::DATE,'1 day'::INTERVAL) datum
